    const WORKER_URL = 'https://mypaypicture-api.proud-truth-9c88.workers.dev';
    const SYS_PROMPT = 'You are a helpful UK personal finance assistant for MyPayPicture, a suite of tools helping people understand their pay, pension, and benefits. Help users find the right tool and answer general UK personal finance questions.\n\nTools: MyPayOptimiser (salary sacrifice, contributions), MyPensionReclaim (missed salary sacrifice on bonuses), MyPensionTop-Up (AVCs and tax relief), MyFamily and Benefits (Child Benefit and Tax-Free Childcare), MyRetirementHelper (State Pension, drawdown, sustainability).\n\nRules:\n- UK English, 2024/25 rates\n- General information only, not personalised financial advice\n- When recommending a tool, always use its exact name: MyPayOptimiser, MyPensionReclaim, MyPensionTop-Up, MyRetirementHelper so users can click through\n- Direct users to the right tool\n- Warm, clear, approachable tone';
    let chatHist = [];
    let chatBusy = false;
    function chatAddMsg(role, text, typing) {
    const msgs = document.getElementById('chatMsgs');
    const div = document.createElement('div');
    div.className = 'chat-msg ' + (role === 'user' ? 'user' : 'ai');
    if (typing) div.id = 'chatTyping';
    const av = role === 'user' ? 'You' : '✦';
    let bubble;
    if (typing) {
    bubble = '<div class="chat-typing"><span></span><span></span><span></span></div>';
    } else {
    const toolLinks = {
    'MyPayOptimiser': 'pay-optimiser.html',
    'MyPensionReclaim': 'pension-reclaim.html',
    'MyPensionTop-Up': 'pension-topup.html',
    'MyRetirementHelper': 'retirement-helper.html',
    };
    const formatted = text.split('\n\n').map(function(p) {
    var line = p.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>');
    Object.keys(toolLinks).forEach(function(name) {
    line = line.split(name).join('<a href="' + toolLinks[name] + '" style="color:var(--navy,#0f1f38);font-weight:600;text-decoration:underline;">' + name + '</a>');
    });
    return line;
    }).join('</p><p>');
    // Render markdown links [text](url)
    const linkedFormatted = formatted.replace(/\[([^\]]+)\]\((https?:\/\/[^)]+)\)/g, '<a href="$2" target="_blank" style="color:var(--navy,#0f1f38);font-weight:600;text-decoration:underline;">$1</a>');
    bubble = '<p>' + linkedFormatted + '</p>';
    }
    div.innerHTML = '<div class="msg-av">' + av + '</div><div class="msg-bub">' + bubble + '</div>';
    msgs.appendChild(div);
    msgs.scrollTop = msgs.scrollHeight;
    }
    async function chatSend() {
    const input = document.getElementById('chatInput');
    const btn = document.getElementById('chatSendBtn');
    const text = input.value.trim();
    if (!text || chatBusy) return;
    document.getElementById('chatChips').style.display = 'none';
    chatAddMsg('user', text);
    chatHist.push({ role: 'user', content: text });
    input.value = '';
    input.style.height = 'auto';
    chatBusy = true;
    btn.disabled = true;
    chatAddMsg('ai', '', true);
    try {
    const res = await fetch(WORKER_URL, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ messages: chatHist, system: SYS_PROMPT }),
    });
    const data = await res.json();
    const reply = (data.content && data.content[0] && data.content[0].text)
    ? data.content[0].text
    : (data.error || 'Sorry, something went wrong. Please try again.');
    var t = document.getElementById('chatTyping');
    if (t) t.remove();
    if (isComplex) {
    const ctaNote = '\n\n---\n**Want personalised advice?** A regulated financial adviser can help with your specific situation. [Speak to an adviser →](https://tally.so/r/xX2YP5)';
    chatAddMsg('ai', reply + ctaNote);
    } else {
    chatAddMsg('ai', reply);
    }
    // Show adviser CTA after any response
    showAdviserCta();
    chatHist.push({ role: 'assistant', content: reply });
    } catch (err) {
    var t = document.getElementById('chatTyping');
    if (t) t.remove();
    chatAddMsg('ai', 'Unable to connect. Please check your connection and try again.');
    }
    chatBusy = false;
    btn.disabled = false;
    input.focus();
    }
    function chatChip(el) {
    document.getElementById('chatInput').value = el.textContent;
    chatSend();
    }
